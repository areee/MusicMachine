7   /**
8    * Tekstikäyttöliittymä, jolla valitaan, halutaanko avata MusicMachinen
9    * graafinen vai tekstipohjainen käyttöliittymä
10   */
11  public class ValitseKayttoliittyma {
12  
13      private final Lukija lukija;
14  
15      /**
16       * Asettaa lukijaksi uuden Lukija-luokan
17       */
18      public ValitseKayttoliittyma() {
19          lukija = new Lukija();
20      }
21  
22      /**
23       * Käynnistää käyttäjän valinnan mukaan joko tekstipohjaisen tai graafisen
24       * käyttöliittymän
25       *
26       * @throws java.io.IOException
27       */
28      public void kaynnista() throws IOException {
29  
30  //        OUTER:
31          while (true) {
32              System.out.println("Valitse käyttöliittymä:\n"
33                      + "1) Graafinen käyttöliittymä\n"
34                      + "2) Tekstikäyttöliittymä\n"
35                      + "x) Lopeta");
36              System.out.print(lukija);
37              String komento = lukija.annaKomento();
38  
39              if (komento.equals("1")) {
40                  new GraafinenKayttoliittyma().setVisible(true);
41                  break;
42  
43              } else if (komento.equals("2")) {
44                  new TekstiKayttoliittyma().kaynnista();
45                  break;
46              } else if (komento.equals("x")) {
47                  break;
48              }
49          }
50      }
51  }
